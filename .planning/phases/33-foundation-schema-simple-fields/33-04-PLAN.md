<!-- use-vehicles.ts query hook pattern (to replicate for vehicle-classes) -->
From apps/web/src/hooks/queries/use-vehicles.ts:
```typescript
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { apiClient } from '@/lib/api-client';
import { toast } from 'sonner';
export const vehicleKeys = {
  all: ['vehicles'] as const,
  list: (filters?: Record<string, unknown>) => [...vehicleKeys.all, 'list', filters] as const,
  detail: (id: string) => [...vehicleKeys.all, 'detail', id] as const,
};
export function useVehicles(filter) {
  return useQuery({ queryKey: vehicleKeys.list({ filter }), queryFn: () => apiClient<VehicleDto[]>(`/vehicles?filter=${filter}`) });
}
export function useCreateVehicle() {
  const queryClient = useQueryClient();
  return useMutation({
    mutationFn: (data) => apiClient<VehicleDto>('/vehicles', { method: 'POST', body: JSON.stringify(data) }),
    onSuccess: () => { queryClient.invalidateQueries({ queryKey: vehicleKeys.all }); toast.success('Pojazd dodany pomyslnie'); },
    onError: () => { toast.error('Wystapil blad...'); },
  });
}
```

- GET /vehicles includes vehicleClass: { id, name }
