    1. Create `apps/api/src/mail/mail.service.ts`:
       ```typescript
       import { Injectable } from '@nestjs/common';
       import { ConfigService } from '@nestjs/config';
       import * as nodemailer from 'nodemailer';

       @Injectable()
       export class MailService {
         private transporter: nodemailer.Transporter;

         constructor(private config: ConfigService) {
           this.transporter = nodemailer.createTransport({
             host: this.config.get('MAIL_HOST'),
             port: this.config.get<number>('MAIL_PORT'),
             secure: false,
           });
         }

         async sendSetupPasswordEmail(to: string, name: string, token: string): Promise<void> {
           const appUrl = this.config.get('APP_URL');
           const link = `${appUrl}/setup-password?token=${token}`;
           await this.transporter.sendMail({
             from: this.config.get('MAIL_FROM'),
             to,
             subject: 'RentApp - Ustaw haslo do konta',
             html: `<p>Czesc ${name},</p><p>Twoje konto zostalo utworzone. Kliknij ponizszy link aby ustawic haslo:</p><p><a href="${link}">${link}</a></p><p>Link wygasa za 72 godziny.</p>`,
           });
         }

         async sendResetPasswordEmail(to: string, name: string, token: string): Promise<void> {
           const appUrl = this.config.get('APP_URL');
           const link = `${appUrl}/reset-password?token=${token}`;
           await this.transporter.sendMail({
             from: this.config.get('MAIL_FROM'),
             to,
             subject: 'RentApp - Reset hasla',
             html: `<p>Czesc ${name},</p><p>Kliknij ponizszy link aby zresetowac haslo:</p><p><a href="${link}">${link}</a></p><p>Link wygasa za 1 godzine.</p>`,
           });
         }
       }
       ```

    4. Create `apps/api/src/users/users.service.ts`:
       - `createUser(dto: CreateUserDto, adminId: string)`: Creates user with `passwordHash: null`, generates 32-byte hex setup token via `crypto.randomBytes(32).toString('hex')`, hashes with argon2 (`{ memoryCost: 32768, timeCost: 3, parallelism: 1 }`), stores hash in `setupToken`, sets `setupTokenExpiry` to `Date.now() + 72h`, calls `mailService.sendSetupPasswordEmail()` with raw token, returns `{ id, email, name, role }`
       - `findByEmail(email: string)`: Returns user or null
       - `findById(id: string)`: Returns user or null
       - `requestPasswordReset(email: string)`: Finds user by email, generates reset token (same mechanism as setup token but 1h expiry), calls `mailService.sendResetPasswordEmail()`, returns void (does NOT reveal if email exists -- silently returns if user not found)
       - Import argon2 with `import * as argon2 from 'argon2'`
       - Import crypto with `import * as crypto from 'crypto'`
