import { IdCardOcrResult } from '@rentapp/shared';

export function parseIdCard(ocrTexts: string[]): IdCardOcrResult {
  const fullText = ocrTexts.join('\n');

  // PESEL: exactly 11 consecutive digits (not part of longer number)
  const peselMatch = fullText.match(/(?<!\d)\d{11}(?!\d)/);

  // Document number: 3 uppercase letters + 6 digits (e.g., ABC123456)
  const docNumMatch = fullText.match(/[A-Z]{3}\d{6}/);

  // Name extraction: filter out header/label lines, find name-like lines
  // Polish ID has labels in Polish and English; names are standalone lines with Polish characters
  const nameLines = ocrTexts
    .map(line => line.trim())
    .filter(line =>
      line.length >= 2 &&
      line.length <= 40 &&
      !line.match(/RZECZPOSPOLITA|DOWÓD|DOWOD|POLSKA|POLAND|IDENTITY|CARD|KARTA|Surname|Name|Nazwisko|Imiona|Date|Data|Nationality|Obywatelstwo|Sex|Plec|PESEL|Nr\s*dok/i) &&
      line.match(/^[A-ZĄĆĘŁŃÓŚŹŻ][a-ząćęłńóśźż\-]+$/)
    );

  return {
    firstName: nameLines.length > 1 ? nameLines[1] : null,
    lastName: nameLines.length > 0 ? nameLines[0] : null,
    pesel: peselMatch?.[0] ?? null,
    documentNumber: docNumMatch?.[0] ?? null,
  };
}
```

import { DriverLicenseOcrResult } from '@rentapp/shared';

export function parseDriverLicense(ocrTexts: string[]): DriverLicenseOcrResult {
  const fullText = ocrTexts.join('\n');

  // License number: Polish format XXXXX/XX/XXXX (5 digits / 2 digits / 4 digits)
  // Also try alternative formats: digits with slashes
  const licenseNumMatch = fullText.match(/\d{5}\/\d{2}\/\d{4}/) 
    ?? fullText.match(/\d{4,6}\/\d{2}\/\d{2,4}/);

  // Categories: standard EU categories
  const categoryPattern = /(?:AM|A1|A2|A|B1|B\+E|BE|B|C1\+E|C1E|C1|C\+E|CE|C|D1\+E|D1E|D1|D\+E|DE|D|T)(?:\s*[,\s]\s*(?:AM|A1|A2|A|B1|B\+E|BE|B|C1\+E|C1E|C1|C\+E|CE|C|D1\+E|D1E|D1|D\+E|DE|D|T))*/;
  const categoryMatch = fullText.match(categoryPattern);

  // Dates: DD.MM.YYYY or DD/MM/YYYY
  const dates = fullText.match(/\d{2}[./]\d{2}[./]\d{4}/g) ?? [];
  // Sort by position -- last date is typically expiry (field 4b comes after 4a)
  const expiryRaw = dates.length > 1 ? dates[dates.length - 1] : dates[0];

  let expiryDate: string | null = null;
  if (expiryRaw) {
    const parts = expiryRaw.split(/[./]/);
    if (parts.length === 3) {
      expiryDate = `${parts[2]}-${parts[1]}-${parts[0]}`; // ISO format
    }
  }

  return {
    licenseNumber: licenseNumMatch?.[0] ?? null,
    categories: categoryMatch?.[0]?.replace(/\s+/g, ', ') ?? null,
    expiryDate,
  };
}
```
