1. Update `.github/workflows/deploy-api.yml` -- replace the "Wait for deployment" step (sleep 30) and "Health check" step with a single polling step:
```yaml
- name: Health check (polling)
  env:
    API_URL: ${{ vars.API_URL }}
  run: |
    if [ -z "$API_URL" ]; then
      echo "API_URL not configured - skipping health check"
      exit 0
    fi
    for i in $(seq 1 12); do
      STATUS=$(curl -s -o /dev/null -w '%{http_code}' "${API_URL}/health" || echo "000")
      if [ "$STATUS" = "200" ]; then
        echo "Health check passed on attempt $i"
        exit 0
      fi
      echo "Attempt $i/12: status $STATUS - retrying in 10s"
      sleep 10
    done
    echo "Health check failed after 12 attempts (2 minutes)"
    exit 1
```

Remove the `sleep 30` step entirely. The polling loop waits up to 2 minutes with 10s intervals.

2. Update `.github/workflows/deploy-web.yml` -- add a health check step after the deploy step:
```yaml
- name: Health check (polling)
  env:
    WEB_URL: ${{ vars.WEB_URL }}
  run: |
    if [ -z "$WEB_URL" ]; then
      echo "WEB_URL not configured - skipping health check"
      exit 0
    fi
    for i in $(seq 1 12); do
      STATUS=$(curl -s -o /dev/null -w '%{http_code}' "${WEB_URL}/api/health" || echo "000")
      if [ "$STATUS" = "200" ]; then
        echo "Health check passed on attempt $i"
        exit 0
      fi
      echo "Attempt $i/12: status $STATUS - retrying in 10s"
      sleep 10
    done
    echo "Health check failed after 12 attempts (2 minutes)"
    exit 1
```
